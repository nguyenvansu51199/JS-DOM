Simple selectors: 					bộ chọn đơn giản
	*						Chọn tất cả thẻ
	element.class		p.intro			Chọn tất cả thẻ <p> có class intro
	element, element		div, p			Chọn tất cả thẻ <div> và tất cả thẻ <p>
	.class1.class2		.name1.name2		Chọn tất cả các thẻ có cả class name1 và name2

Pseudo-classes selector
	:hover			p:hover			Chọn các thẻ <p> khi di chuột qua
	Đi kèm với :hover thì nên sử dụng thêm thuộc tính "transition"

	:active			a:active			Chọn các thẻ <a> khi nhấn chuột xuống
	:link			a:link			Chọn các thẻ <a> có liên kết chưa truy cập
	:visited			a:visited			Chọn các thẻ <a> có liên kết đã truy cập
	:empty			div:empty			Chọn các thẻ <div> không có nội dung
	:target			#home:target		Chọn thẻ có id="home" khi trên URL có hash tương ứng

Type selectors
	:first-of-type		p:first-of-type		Chọn các thẻ <p> đầu tiên trong thẻ cha
	:last-of-type		p:last-of-type		Chọn các thẻ <p> cuối cùng trong thẻ cha
	:nth-of-type(n)		p:nth-of-type(n)		Chọn các thẻ <p> thứ hai trong thẻ cha		.Giá trị n: số tự nhiên, odd(lẻ), even(chẳn)
	:nth-last-of-type(n)		p:nth-last-of-type(n)		Chọn các thẻ <p> thứ hai trong thẻ cha, tính từ cuối lên
	:only-of-type		p:only-of-type		Chọn các thẻ <p> duy nhất trong thẻ cha

Child selectors
	:first-child		p:first-child		Chọn các thẻ <p> là thẻ con đầu tiên
	:last-child		p:last-child		Chọn các thẻ <p> là thẻ con cuối cùng
	:nth-child(n)	p:nth-child(n)	Chọn các thẻ <p> là thẻ con thứ hai
	:nth-last-child(n)	p:nth-last-child(n)	Chọn các thẻ <p> là thẻ con thứ hai, tính từ cuối lên
	:only-child		p:only-child		Chọn các thẻ <p> là thẻ con duy nhất

Combinator selectors
	element element	div p			Chọn tất cả thẻ <p> nằm trong thẻ <div>
	element > element	div > p			Chọn tất cả thẻ <p> là con trực tiếp của thẻ <div>
			div > p > i			Chọn tất cả  thẻ <i> là con trực tiếp của thẻ <p>. Với điều kiện thẻ <p> là con trực tiếp của thẻ <div>
	element + element	div + p			Chọn một thẻ <p> đầu tiên đứng ngay sau thẻ <div>
			p + p			Chọn các thẻ <p> đứng đằng sau thẻ <p> trừ thẻ <p> đầu tiên			
	element1 ~ element2	div ~ p			Chọn tất cả thẻ <p> đứng sau thẻ <div>

Attribute selectors
	[attribute] 		[href]			Chọn tất cả các thẻ có thuộc tính href
			a[href]			Chọn tất cả các thẻ a có thuộc tính href
			.link[href]			Chọn tất cả các thẻ có class="link" có thuộc tính href
	[attribute=value] 	[lang=vi]			Chọn tất cả các thẻ có thuộc tính lang chỉ có giá trị bằng vi
			a[href=#!]			Chọn tất cả các thẻ a có thuộc tính href chỉ có giá trị bằng #!
			#link[href=#]		Chọn tất cả các thẻ có id="link" có thuộc tính href chỉ có giá trị bằng #
	[attribute~=value] 	[title~=sot]			Chọn tất cả các thẻ có thuộc tính chứa từ sot. Thường dùng trên thuộc tính title
			[title~=sot][title~=tin]		Chọn tất cả các thẻ có thuộc tính chứa từ sot và từ tin. Thường dùng trên thuộc tính title
	[attribute|=value]	[lang|=en]			Chọn tất cả cấc thẻ có thuộc tính lang có giá trị bằng en hoặc giá trị bắt đầu bằng en- 
						Thường dùng ở thuộc tính lang=""(ngôn ngữ)
	[attribute^=value]	a[href^="https://"]		Chọn tất cả các thẻ có thuộc tính href giá trị bắt đầu bằng https://
						Thường dùng ở thẻ a, liên kết
	[attribute$=value]	a[href$=".pdf"]		Chọn tất cả các thẻ có thuộc tính href giá trị kết thúc bằng .pdf
						Thường dùng ở link, hình ảnh
	[attribute*=value]	a[href*="w3schools"]		Chọn tất cả các thẻ có thuộc tính href chứa giá trị bằng w3schools
